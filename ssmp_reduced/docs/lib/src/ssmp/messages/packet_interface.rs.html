// ============================ Raw Byte Parsing =========================== //
/// # PacketParseStatus Enum
/// This enum defines all possible parse successes and errors that could be
/// returned from a PacketParser parsing function.
#[allow(non_camel_case_types)]
#[derive(Copy)]
pub enum PacketParseStatus {
    // ----- Success Messages ------ //
    SUCCESS, // a generic success indicator used to describe a
    // success parse
    // ----------------------------- //
    _STATUS_DELIMETER, // used internally to serve as a delimeter between
    // success and error messages
    // ------ Error Messages ------- //
    ERROR_TOO_FEW_BYTES, // not enough bytes are supplied to fill a packet
    // header. Thus, a packet couldn't be created
    ERROR_NO_HEADER, // `parse_data()` was called, but the header hasn't
    // been parsed yet
    ERROR_NO_DATA_IMPLIED, // `parse_data()` was called, but the header's
    // fields imply there is no data expected
    ERROR_DATA_APPEND, // attempting to append data to the packet's user
                       // header or payload failed.
}

// Implementing the 'clone' function so the enum can be copied. This makes
// working with the enum much easier
impl Clone for PacketParseStatus {
    fn clone(&self) -> PacketParseStatus {
        *self
    }
}
impl PacketParseStatus {
    /// Returns whether or not the given status is a success message
    pub fn is_success(&self) -> bool {
        return (*self as u32) < (PacketParseStatus::_STATUS_DELIMETER as u32);
    }

    /// Returns whether or not the given status is an error message
    pub fn is_error(&self) -> bool {
        return (*self as u32) > (PacketParseStatus::_STATUS_DELIMETER as u32);
    }
}

/// # Packet Parser
/// The `PacketParser` struct is responsible for taking in a collection of bytes
/// and parsing it into a `Packet` struct instance.
pub struct PacketParser {
    data: Vec<u8>,      // holds raw byte data
    pub packet: Packet, // holds the parsed Packet struct
    byte_index: u32,    // represents the current byte index the parser is on
                        // within the packet
}

impl PacketParser {
    /// **Default constructor**: takes in nothing and returns a default
    /// `PacketParser` instance. This is useful when needing to slowly build a
    /// `Packet` over time by slowly giving bytes to the `PacketParser`.
    pub fn new() -> PacketParser {
        return PacketParser {
            data: Vec::new(),
            packet: Packet::new(),
            byte_index: 0,
        };
    }

    // ----------------- Byte Parsing ------------------ //
    /// This is the first function to be called when parsing a new packet. Its
    /// job is to parse the header segment of a packet.
    ///
    /// It takes in a vector of bytes (which must be *at least* 12 in length)
    /// and attempts to parse the values of the packet header. If the vector
    /// is:
    ///
    /// - Shorter than 12 bytes: `PacketParseStatus::ERROR_TOO_FEW_BYTES`
    ///   is returned
    /// - Longer than 12 bytes: only the first 12 bytes will be used
    ///
    /// On a successful parse, this `PacketParser`'s `packet` field will hold
    /// the header values parsed from the given bytes.
    ///
    /// **NOTE**: This function assumes a new packet is being read and wipes
    /// any previous packet information stored from a previous parse. However,
    /// if not enough bytes are supplied, and
    /// `PacketParseStatus::ERROR_TOO_FEW_BYTES` is returned, the previous
    /// information will *not* be wiped.
    pub fn parse_header(&mut self, bytes: &mut Vec<u8>) -> PacketParseStatus
    {
        // check the length of the byte vector and return an error if there
        // aren't enough bytes to form a packet header
        if bytes.len() < 12 {
            return PacketParseStatus::ERROR_TOO_FEW_BYTES;
        }

        // wipe the previous information stored in the PacketParser
        self.reset();
        self.data.append(&mut bytes[0..12].to_vec());

        // ------- 8-bit Header Fields ------- //
        // fill the packet's header fields and increment the byte index
        self.packet.header.magic = self.data[0];
        self.packet.header.version = self.data[1];
        self.packet.header.control = PacketControl::new(self.data[2]);
        self.packet.header.checksum = self.data[3];
        self.byte_index += 4;

        // ------ 32-bit Header Fields ------- //
        // attempt to convert the next 8 bytes to two 32-bit values
        let mut header_32bit_fields: Vec<u32> = Vec::with_capacity(2);
        for _i in 0..2 {
            // grab a slice of 4 bytes from the current byte index
            let byte_range = (self.byte_index as usize)..((self.byte_index + 4) as usize);
            let bytes: Vec<u8> = self.data[byte_range].to_vec();
            // use the toolbox function to convert and store in the temp vector
            header_32bit_fields.push(packet_toolbox::byte_vector_to_u32(bytes));
            self.byte_index += 4; // increment byte index 4 more bytes
        }
        // save off the two 32-bit values to the appropriate header fields
        self.packet.header.packet_length = header_32bit_fields[0];
        self.packet.header.payload_offset = header_32bit_fields[1];

        // return a success status message (indicating that only the header
        // fields of the packet were filled
        return PacketParseStatus::SUCCESS;
    }

    /// This is the second function to be called when parsing a new
    /// packet. It's job is to parse the user header and payload segments of a
    /// SSMP packet. (It should only be called if the `packet_length` implies
    /// there is a user header and/or payload.)
    ///
    /// It takes in a vector of bytes (which should be *at least* the expected
    /// length (`self.packet.header.packet_length - 12`)) and attempts to parse
    /// out the user header and payload. If the vector is:
    ///
    /// - Shorter than expected: `PacketParseStatus::ERROR_TOO_FEW_BYTES`
    ///   is returned
    /// - Longer than expected: only the first 12 bytes will be used
    ///
    /// On a successful parse, a success `PacketParseStatus` will be returned.
    /// If this function is called and the packet's header values imply there
    /// is NO user header or payload, this function will return
    /// `PacketParseStatus::ERROR_NO_DATA_IMPLIED`
    ///
    /// **NOTE**: This function expects `self.parse_header()` to have already
    /// been executed. If it hasn't, this function will return
    /// `PacketParseStatus::ERROR_NO_HEADER`.
    pub fn parse_data(&mut self, bytes: &mut Vec<u8>) -> PacketParseStatus
    {
        // check to ensure the header has been filled:
        if self.byte_index < 12 {
            return PacketParseStatus::ERROR_NO_HEADER;
        }

        // calculate the expected data length. If it's 0, then there's no
        // data expected. Return from this function with an appropriate error
        let data_length: u32 = self.packet.header.packet_length - 12;
        if data_length == 0 {
            return PacketParseStatus::ERROR_NO_DATA_IMPLIED;
        }

        // if the number of bytes supplied is less than the expected amount
        // (which we can determine using the `packet_length` field), return
        // an error
        if (bytes.len() as u32) < data_length {
            return PacketParseStatus::ERROR_TOO_FEW_BYTES;
        }

        // append the exact number of bytes needed to fill the user header
        // and/or payload to the PacketParser's (this will exclude any extra
        // bytes at the end of the vector)
        self.data
            .append(&mut bytes[0..(data_length as usize)].to_vec());

        // calculate ranges over the data vector that represent each segment
        let uheader_range = 12..(self.packet.header.payload_offset as usize);
        let payload_range = (self.packet.header.payload_offset as usize)
            ..(self.packet.header.packet_length as usize);
        // make copies of slices over these ranges
        let mut uheader_data: Vec<u8> = self.data[uheader_range].to_vec();
        let mut payload_data: Vec<u8> = self.data[payload_range].to_vec();

        // before appending to the user header and payload, we need to subtract
        // the values that are about to be added to the `packet_length` and
        // `payload_offset` fields. This is done because the append() functions
        // called below will do this automatically, despite the fact that these
        // header fields are already set. Subtracting will keep the two fields
        // at their original value. These corrections are below, and are
        // marked with two stars (/**/)

        // subtract from the header fields to account for the double-addition
        // described above, then attempt to append to the user header
        let uheader_length: u32 = self.packet.header.payload_offset - 12; /**/
        self.packet.header.payload_offset -= uheader_length; /**/
        self.packet.header.packet_length -= uheader_length; /**/
        let uh_result: bool = self.packet.user_header_append_bytes(&mut uheader_data);

        // reset the packet length field before appending to the payload (to
        // account for the double-addition), then attempt to append to the user
        // header
        self.packet.header.packet_length = self.packet.header.payload_offset; /**/
        let pl_result: bool = self.packet.payload_append_bytes(&mut payload_data);

        // if either append fails, return an appropriate error status
        if !(uh_result && pl_result) {
            return PacketParseStatus::ERROR_DATA_APPEND;
        }

        // return a success status
        return PacketParseStatus::SUCCESS;
    }

    // --------------- Utility Functions --------------- //
    /// Reset the parsed packet (and any internal fields) so that the same
    /// `PacketParser` object may be used to parse another packet
    pub fn reset(&mut self) {
        self.data = Vec::new(); // wipe data vector
        self.packet = Packet::new(); // wipe packet
        self.byte_index = 0; // zero-out byte index
    }
}

    // Tests parsing ONLY the header of a packet, given 12 or more bytes
    #[test]
    pub fn packet_interface_test1_parse_header() {
        // -------------- TEST 0: less than 12 bytes ------------- //
        let mut raw_bytes: Vec<u8> = vec![0xdb, 0xaa, 0xbb];
        let mut pp0: PacketParser = PacketParser::new();
        let pp0_result: PacketParseStatus = pp0.parse_header(&mut raw_bytes);
        // make assertions: an error should have been returned
        assert!(pp0_result.is_error());
        assert_eq!(
            pp0_result as u32,
            PacketParseStatus::ERROR_TOO_FEW_BYTES as u32
        );

        // --------------- TEST 1: exactly 12 bytes -------------- //
        let mut raw_bytes: Vec<u8> = vec![0xdb, 0xaa, 0xbb, 0xcc,
                                      0xd1, 0xd2, 0xd3, 0xd4,
                                      0xe1, 0xe2, 0xe3, 0xe4];
        // print raw data
        let lines: Vec<String> =
            packet_toolbox::byte_vector_to_detailed_string_vector(&raw_bytes, 4);
        println!(
            "\n----------\nGiven raw header data ({} bytes):\n",
            raw_bytes.len()
        );
        for l in lines {
            println!("\t{}", l);
        }
        // parse header
        let mut pp1: PacketParser = PacketParser::new();
        let pp1_result: PacketParseStatus = pp1.parse_header(&mut raw_bytes);
        // print resulting packet
        println!("\nParsed packet:\n{}", pp1.packet.to_string());
        // make assertions
        assert!(pp1_result.is_success());
        assert_eq!(pp1.packet.header.magic, 0xdb);
        assert_eq!(pp1.packet.header.version, 0xaa);
        assert_eq!(pp1.packet.header.control.get_raw(), 0xbb);
        assert_eq!(pp1.packet.header.checksum, 0xcc);
        assert_eq!(pp1.packet.header.packet_length, 0xd1d2d3d4);
        assert_eq!(pp1.packet.header.payload_offset, 0xe1e2e3e4);

    // Performs tests with randomized packets
    #[test]
    pub fn packet_interface_test3_thorough() {
        const UHEADER_BYTE: u8 = 0xff;
        const PAYLOAD_BYTE: u8 = 0x00;

        // iterate across uheader_range
        for uheader_length in 0..100 {
            // iterate across payload_range
            for payload_length in 0..100 {
                // determine the packet length
                let packet_length: u32 = uheader_length + payload_length + 12;

                // create a vector of raw bytes
                let mut bytes: Vec<u8> = Vec::with_capacity(12);
                // append header data
                bytes.push(0xdb);
                bytes.push(0x01);
                bytes.push(0x11);
                bytes.push(0xcc);
                // append 32-bit header values
                bytes.push(0x00);
                bytes.push(0x00);
                bytes.push(0x00);
                bytes.push((packet_length) as u8);
                bytes.push(0x00);
                bytes.push(0x00);
                bytes.push(0x00);
                bytes.push((12 + uheader_length) as u8);
                // append data
                let mut data: Vec<u8> = Vec::with_capacity((packet_length - 12) as usize);
                for _b in 0..uheader_length {
                    data.push(UHEADER_BYTE);
                }
                for _b in 0..payload_length {
                    data.push(PAYLOAD_BYTE);
                }

                // parse header
                let mut pp: PacketParser = PacketParser::new();
                assert!(pp.parse_header(&mut bytes).is_success());
                assert_eq!(pp.packet.header.packet_length, packet_length);
                assert_eq!(pp.packet.header.payload_offset, uheader_length + 12);

                // only parse data if there IS data
                if packet_length > 12
                {
                    assert_eq!(pp.parse_data(&mut data) as u32, PacketParseStatus::SUCCESS as u32);
                    assert_eq!(pp.packet.get_user_header_data().len() as u32, uheader_length);
                    assert_eq!(pp.packet.get_payload_data().len() as u32, payload_length);
                }
            }
        }
    }
