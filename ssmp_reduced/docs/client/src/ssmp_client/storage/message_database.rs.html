// A module to hold the struct definitions for interacting with the message
// database. Utilizes `./database.rs`
// 
// Message timestamps make use of the `chrono` library:
// https://docs.rs/chrono/0.4.0/chrono/#structs
//
//      0xdadb0d
//      CS 4284: Systems and Networking Capstone
//      Fall 2020

// Library/module inclusions
use crate::storage::database;
use database::LocalDatabase;
use database::SQLColumnInfo;
use rusqlite::{Connection, Error, NO_PARAMS};
use rusqlite::types::ToSql;
use std::time::{SystemTime, Duration, UNIX_EPOCH};
use chrono::NaiveDateTime;

// NOTE TO SELF: some questions for the group:

// =============================== Constants =============================== //
#[allow(dead_code)]
const MESSAGE_DATABASE_FILE: &str = "messages.db";


// ============================ Message Schema ============================= //
/// # The Message Schema
/// A schema used to organize the data kept in the SQLite message database.
#[derive(Clone)]
pub struct MessageSchema
{
    pub message_id: (i32, SQLColumnInfo),
    pub conversation_id: (i32, SQLColumnInfo),
    pub message: (String, SQLColumnInfo),
    pub sender: (String, SQLColumnInfo),
    pub timestamp: (i64, SQLColumnInfo),
    pub is_read: (bool, SQLColumnInfo)
}

impl MessageSchema
{
    /// **Constructor**: takes in a message (`String`) and a sender (`String`)
    /// and creates a message schema with the other fields filled in with
    /// defaults. The schema's timestamp is set to the current time.
    pub fn new(msg: &str, sender: &str) -> MessageSchema
    {
        // create a message schema
        let ms: MessageSchema = MessageSchema
        {
            message_id: (0, SQLColumnInfo::new("message_id", "INT", true)),
            conversation_id: (0, SQLColumnInfo::new("conversation_id", "INT", false)),
            message: (msg.to_string(), SQLColumnInfo::new("message", "MEDIUMTEXT", false)),
            sender: (sender.to_string(), SQLColumnInfo::new("sender", "MEDIUMTEXT", false)),
            timestamp: (MessageSchema::get_current_datetime_i64(),
                        SQLColumnInfo::new("timestamp", "BIGINT", false)),
            is_read: (false, SQLColumnInfo::new("is_read", "BOOL", false))
        };

        return ms;
    }


    // ------------------------- Utility Functions ------------------------- //
    /// Creates a vector of special `ToSql` values for SQL commands to parse.
    /// The vector contains every struct field in the order they appear.
    #[allow(dead_code)]
    pub fn to_sql_params(&self) -> Vec<&dyn ToSql>
    {
        // create an array with each value
        let result: Vec<&dyn ToSql> = vec![&self.message_id.0 as &dyn ToSql,
                                           &self.conversation_id.0 as &dyn ToSql,
                                           &self.message.0 as &dyn ToSql,
                                           &self.sender.0 as &dyn ToSql,
                                           &self.timestamp.0 as &dyn ToSql,
                                           &self.is_read.0 as &dyn ToSql];
        return result;
    }

    /// Creates and returns a string representation of a `MessageSchema`.
    #[allow(dead_code)]
    pub fn to_string(&self) -> String
    {
        let mut result: String = String::new();
        result.push_str(&format!("{:<18}{}\n", "Message ID:", self.message_id.0));
        result.push_str(&format!("{:<18}{}\n", "Conversation ID:", self.conversation_id.0));
        result.push_str(&format!("{:<18}{}\n", "Sender:", self.sender.0));
        let timestring: String = MessageSchema::datetime_to_string(
                                 MessageSchema::i64_to_datetime(self.timestamp.0));
        result.push_str(&format!("{:<18}{}\n", "Timestamp:", timestring));
        result.push_str(&format!("{:<18}{}\n", "Is Read:", self.is_read.0));
        result.push_str(&format!("{:<18}{}", "Message:", self.message.0));
        // return the end result
        return result;
    }

    /// Takes the schema's timestamp (`i64`) and returns an appropriate
    /// `NaiveDateTime` object.
    #[allow(dead_code)]
    pub fn get_timestamp_as_datetime(&self) -> NaiveDateTime
    {
        return MessageSchema::i64_to_datetime(self.timestamp.0);
    }

    /// Takes the schema's timestamp (`i64`) and returns an appropraite
    /// `NiaveDateTime` object
    #[allow(dead_code)]
    pub fn get_timestamp_as_string(&self) -> String
    {
        return MessageSchema::datetime_to_string(
               MessageSchema::i64_to_datetime(self.timestamp.0));
    }

    /// Returns the SQLColumnInfo of each struct field, all combined in a vec.
    pub fn get_sql_column_info(&self) -> Vec<SQLColumnInfo>
    {
        let mut result: Vec<SQLColumnInfo> = Vec::new();
        result.push(self.message_id.1.clone());
        result.push(self.conversation_id.1.clone());
        result.push(self.message.1.clone());
        result.push(self.sender.1.clone());
        result.push(self.timestamp.1.clone());
        result.push(self.is_read.1.clone());
        return result;
    }


    // ------------------------ Timestamp Functions ------------------------ //
    /// Gets and returns the current system time as a 64-bit unsigned integer.
    fn get_current_datetime_i64() -> i64
    {
        // get the current system time. This returns a result, and will only
        // produce an error if UNIX_EPOCH is *later* than SystemTime::now().
        // I don't forsee this happening, EVER, since our messaging protocol
        // will not be the first thing run on a system, even if that system's
        // time is reset back to January 1st, 1970. So, to save the trouble
        // of an intense workaround, we'll assume it worked, and grab Ok().
        let system_time: Duration = SystemTime::now().duration_since(UNIX_EPOCH).ok().unwrap();
        return system_time.as_secs() as i64;
    }

    /// Takes in a u64 (representing seconds since Jan. 1st, 1970) and returns
    /// a NaiveDateTime object with the proper time set.
    fn i64_to_datetime(seconds: i64) -> NaiveDateTime
    {
        return NaiveDateTime::from_timestamp(seconds, 0);
    }

    /// Takes the given `NaiveDateTime` timestamp and converts it to a readable
    /// string. The string is returned.
    fn datetime_to_string(timestamp: NaiveDateTime) -> String
    {
        let time_string: String = timestamp.format("%Y-%m-%d %H:%M:%S").to_string();
        return time_string;
    }
}


// =========================== Message Database ============================ //
/// # The Message Database
/// The `MessageDatabase` struct defines an interface used to interact directly
/// with a SQLite database storing messages. Each message stored in the
/// database follows the schema defined by `MessageSchema`.
/// 
/// Each `MessageDatabase` struct created will connect to the same SQLite
/// database.
#[allow(dead_code)]
pub struct MessageDatabase
{
    file_path: String,              // path to the .db file
    database: LocalDatabase,        // reference to LocalDatabase
    schema: MessageSchema           // a schema instance, for convenience
}

#[allow(dead_code)]
impl MessageDatabase
{
    /// **Constructor**: creates a new `MessageDatabase` struct
    pub fn new() -> MessageDatabase
    {
        let path_create = database::make_database_path(MESSAGE_DATABASE_FILE);
        if path_create.is_err() // for now, panic on error
        { panic!("Could not get/create SSMP database directory!"); }
        let path: String = path_create.unwrap();

        // create and return a MessageDatabase struct
        return MessageDatabase
        {
            file_path: path.clone(),
            database: LocalDatabase::new(&path),
            schema: MessageSchema::new("", "")
        };
    }

    /// **Constructor**: creates a new `MessageDatabase` struct pointed at a
    /// database file specified by the input parameter, `location`.
    pub fn new_at_location(location: &str) -> MessageDatabase
    {
        // create and return a MessageDatabase struct
        return MessageDatabase
        {
            file_path: location.to_string(),
            database: LocalDatabase::new(location),
            schema: MessageSchema::new("", "")
        };
    }


    // ======================= Database Interaction ======================== //
    // --------------- Table Interaction --------------- //
    /// Attempts to create a new table within the message database. The table's
    /// name is specified by `table_name`. On success, the result contains the
    /// command that was executed. On failure, it contains a SQL error.
    pub fn create_table(&self, table_name: &str) -> Result<String, Error>
    {
        // get a database connection - return error on failure
        let conn: Connection = self.database.get_connection()?;

        // get the schema's column info and pass it to the database.rs
        // function to build a CREATE TABLE command string
        let column_info: Vec<SQLColumnInfo> = self.schema.get_sql_column_info();
        let cmd: String = database::make_create_table_string(table_name, column_info);

        // execute the command
        conn.execute(&cmd, NO_PARAMS)?;
        // close database connection and return the command string
        self.database.close_connection(conn)?;
        return Ok(cmd);
    }

    /// Attempts to delete an existing table within the message database. The
    /// table's name is specified by `table_name`. On success, the result holds
    /// the command string that was executed. On failure, it holds a SQL error
    pub fn drop_table(&self, table_name: &str) -> Result<String, Error>
    {
        // establish a database connection
        let conn: Connection = self.database.get_connection()?;

        // build the DROP TABLE command and execute it
        let cmd: String = database::make_drop_table_string(table_name);
        conn.execute(&cmd, NO_PARAMS)?;

        // close the database connection and return the command string
        self.database.close_connection(conn)?;
        return Ok(cmd);
    }

    // -------------- Inserting Messages --------------- //
    /// Attempts to insert a given message defined by `msg` into the table
    /// whose name is defined by `table_name`.
    pub fn insert_message(&self, table_name: &str, msg: MessageSchema) -> Result<String, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // create the command string with the `MessageSchema`'s column info
        let cinfo: Vec<SQLColumnInfo> = msg.get_sql_column_info();
        let cmd: String = database::make_insert_string(table_name, cinfo);
        
        // execute the command
        conn.execute(&cmd, &msg.to_sql_params())?;
        // close the connection and return the command string
        self.database.close_connection(conn)?;
        return Ok(cmd);
    }

    // --------------- Finding Messages ---------------- //
    /// Searches the database for any messages whose message ID matches the
    /// given ID (`mid`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count` == -1)
    pub fn find_message_by_mid(&self, table_name: &str, max_count: i32, mid: i32) -> Result<Vec<MessageSchema>, Error>
    { return self.find_message(table_name, max_count, &self.schema.message_id.1.name, &mid as &dyn ToSql); }
    
    /// Searches the database for any messages whose conversation ID matches the
    /// given ID (`cid`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count` == -1)
    pub fn find_message_by_cid(&self, table_name: &str, max_count: i32, cid: i32) -> Result<Vec<MessageSchema>, Error>
    { return self.find_message(table_name, max_count, &self.schema.conversation_id.1.name, &cid as &dyn ToSql); }
    
    /// Searches the database for any messages whose sender name matches the
    /// given ID (`sender`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count` == -1)
    pub fn find_message_by_sender(&self, table_name: &str, max_count: i32, sender: &str) -> Result<Vec<MessageSchema>, Error>
    { return self.find_message(table_name, max_count, &self.schema.sender.1.name, &sender as &dyn ToSql); }
    
    /// Searches the database for any messages whose message matches the
    /// given ID (`msg`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count` == -1)
    pub fn find_message_by_message(&self, table_name: &str, max_count: i32, msg: &str) -> Result<Vec<MessageSchema>, Error>
    { return self.find_message(table_name, max_count, &self.schema.message.1.name, &msg as &dyn ToSql); }

    /// Private helper that searches the database using ANY MessageSchema field.
    /// The field used to search must be properly casted to a `&dyn ToSql`.
    /// For example:
    /// 
    /// To search by message ID:
    /// 
    /// ```
    /// find_message(table_name, max_count, "message_id",
    ///              &schema.message_id.0 as &dyn ToSql);
    /// ```
    /// 
    /// To search by message content:
    /// 
    /// ```
    /// find_message(table_name, max_count, "conversation_id",
    ///              &schema.message.0 as &dyn ToSql);
    /// ```
    fn find_message(&self, table_name: &str, max_count: i32,
                           field_name: &str, val: &dyn ToSql) -> Result<Vec<MessageSchema>, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // craft a command string with the given information
        let condition: String = format!("{} = ?1", field_name);
        let mut cmd: String = database::make_select_string(table_name, &condition,
                                                           self.schema.get_sql_column_info());
        // add an 'ORDER BY' portion of the string (we want to query for
        // the most recent messages)
        cmd.push_str(&format!(" ORDER BY {} DESC;", self.schema.message_id.1.name));

        // prepare the command for execution, then perform a query map on it
        // (using our `val` input parameter)
        let mut command = conn.prepare(&cmd)?;
        let iterator = command.query_map(&[&val], |row|
        {
            // create a message schema to store this row's fields
            let mut ms: MessageSchema = MessageSchema::new("", "");
            ms.message_id.0 = row.get(0)?;          // message_id
            ms.conversation_id.0 = row.get(1)?;     // conversation_id
            ms.message.0 = row.get(2)?;             // message
            ms.sender.0 = row.get(3)?;              // sender
            ms.timestamp.0 = row.get(4)?;           // timestamp
            ms.is_read.0 = row.get(5)?;             // is_read
            // return the message schema
            return Ok(ms);
        })?;

        // use the above iterator to build a vector of schema
        let mut result: Vec<MessageSchema> = Vec::new();
        let mut schema_count: i32 = 0;
        for ms in iterator
        {
            // push the message schema onto the vector
            result.push(ms.unwrap());
            // increment our counter and ensure we haven't hit our max_count
            schema_count += 1;
            if max_count > 0 && schema_count == max_count
            { break; }
        }

        // return the vector
        //self.database.close_connection(conn)?;
        return Ok(result);
    }

    /// Attempts to retrieve ALL messages in the given table (update to
    /// `max_count` amount).
    pub fn find_all(&self, table_name: &str, max_count: i32) -> Result<Vec<MessageSchema>, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // put together a command string (and add an 'ORDER BY' portion of the
        // string (we want to query for the most recent messages)
        let mut cmd: String = format!("SELECT * FROM {}", table_name);
        cmd.push_str(&format!(" ORDER BY {} DESC;", self.schema.message_id.1.name));

        // prepare the command for execution, then perform a query map on it
        // (using our `val` input parameter)
        let mut command = conn.prepare(&cmd)?;
        let iterator = command.query_map(NO_PARAMS, |row|
        {
            // create a message schema to store this row's fields
            let mut ms: MessageSchema = MessageSchema::new("", "");
            ms.message_id.0 = row.get(0)?;          // message_id
            ms.conversation_id.0 = row.get(1)?;     // conversation_id
            ms.message.0 = row.get(2)?;             // message
            ms.sender.0 = row.get(3)?;              // sender
            ms.timestamp.0 = row.get(4)?;           // timestamp
            ms.is_read.0 = row.get(5)?;             // is_read
            // return the message schema
            return Ok(ms);
        })?;

        // use the above iterator to build a vector of schema
        let mut result: Vec<MessageSchema> = Vec::new();
        let mut schema_count: i32 = 0;
        for ms in iterator
        {
            // push the message schema onto the vector
            result.push(ms.unwrap());
            // increment our counter and ensure we haven't hit our max_count
            schema_count += 1;
            if max_count > 0 && schema_count == max_count
            { break; }
        }

        // return the vector
        //self.database.close_connection(conn)?;
        return Ok(result);
    }


    // --------------- Deleting Messages --------------- //
    /// Searches the database for any messages whose message ID matches the
    /// given ID (`mid`). If any are found, they're deleted from the table.
    /// Returns the number of items deleted, or `rusqlite::Error` on error.
    pub fn delete_message_by_mid(&self, table_name: &str, mid: i32) -> Result<usize, Error>
    { return self.delete_message(table_name, &self.schema.message_id.1.name, &mid as &dyn ToSql); }
    
    /// Searches the database for any messages whose conversation ID matches the
    /// given ID (`cid`). If any are found, they're deleted from the table.
    /// Returns the number of items deleted, or `rusqlite::Error` on error.
    pub fn delete_message_by_cid(&self, table_name: &str, cid: i32) -> Result<usize, Error>
    { return self.delete_message(table_name, &self.schema.conversation_id.1.name, &cid as &dyn ToSql); }
    
    /// Searches the database for any messages whose sender name matches the
    /// given ID (`sender`). If any are found, they're deleted from the table.
    /// Returns the number of items deleted, or `rusqlite::Error` on error.
    pub fn delete_message_by_sender(&self, table_name: &str, sender: &str) -> Result<usize, Error>
    { return self.delete_message(table_name, &self.schema.sender.1.name, &sender as &dyn ToSql); }
    
    /// Searches the database for any messages whose message matches the
    /// given ID (`msg`). If any are found, they're deleted from the table.
    /// Returns the number of items deleted, or `rusqlite::Error` on error.
    pub fn delete_message_by_message(&self, table_name: &str, msg: &str) -> Result<usize, Error>
    { return self.delete_message(table_name, &self.schema.message.1.name, &msg as &dyn ToSql); }

    

    /// Private helper that deletes from the database using ANY MessageSchema
    /// field. The field must be properly casted to a `&dyn ToSql` when being
    /// passed in. For example:
    /// 
    /// To delete by message ID:
    /// 
    /// ```
    /// delete_message(table_name, "message_id",
    ///                &schema.message_id.0 as &dyn ToSql);
    /// ```
    /// 
    /// To delete by conversation ID:
    /// 
    /// ```
    /// delete_message(table_name, "conversation_id",
    ///                &schema.conversation_id.0 as &dyn ToSql);
    /// ```
    fn delete_message(&self, table_name: &str, field_name: &str,
                      val: &dyn ToSql) -> Result<usize, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // craft a command string with the given information
        let condition: String = format!("{} = ?1", field_name);
        let cmd: String = database::make_delete_string(table_name, &condition);
        // execute the command (with `val` used as the parameter)
        let num_deleted = conn.execute(&cmd, &[&val])?;

        // close the connection and return
        self.database.close_connection(conn)?;
        return Ok(num_deleted);
    }

    
    // --------------- Updating Messages --------------- //
    /// Takes in a message ID and a boolean and attempts to update the message
    /// with the given `mid`'s `is_read` flag to the given boolean.
    /// Returns the number of updated elements on success and `rusqlite::Error`
    /// on failure.
    pub fn update_message_is_read(&self, table_name: &str, mid: i32, is_read: bool) -> Result<usize, Error>
    {
        return self.update_message(table_name, vec![&self.schema.is_read.1.name],
                                   vec![&is_read as &dyn ToSql],
                                   &self.schema.message_id.1.name,
                                   &mid as &dyn ToSql);
    }

    /// An in-depth private helper function that takes in values to update
    /// entries within the database. The parameters are as follows:
    /// 
    /// - `table_name` - the name of the table whose values will be updated
    /// - `update_cols` - the names of the columns in the selected entries that
    ///   will be updated
    /// - `update_vals` - a vector of `&dyn ToSql` values used to update the
    ///   column values corresponding to `update_cols`
    /// - `field_name` - the field used to narrow down search results
    /// - `val` - the value matched up with `field_name` to narrow down search
    fn update_message(&self, table_name: &str, update_cols: Vec<&str>,
                      update_vals: Vec<&dyn ToSql>, field_name: &str,
                      val: &dyn ToSql) -> Result<usize, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // create condition substring for the final command string
        let condition: String = format!("{} = ?1", field_name);

        // make sure the two vectors are of the same length
        if update_cols.len() != update_vals.len()
        { return Err(Error::InvalidParameterCount(update_cols.len(), update_vals.len())); }
        // if they are, proceed to build the 'update' portion of the string
        let mut update: String = String::new();
        for i in 0..update_cols.len()
        {
            // the '?{}' parameters need to start at ?2, since the condition
            // string (above) is using ?1
            update.push_str(&format!("{} = ?{}", update_cols[i], i + 2));
            // add a comma if necessary
            if i < update_cols.len() - 1
            { update.push_str(", "); }
        }

        // create a vector containing BOTH `val` and the contents of
        // `update_vals`
        let mut arg_vec: Vec<&dyn ToSql> = vec![&val];
        arg_vec.append(&mut update_vals.clone());
        
        // build the final command string and execute it
        let cmd: String = format!("UPDATE {} SET {} WHERE {};",
                                  table_name, update, condition);
        let num_updated = conn.execute(&cmd, &arg_vec)?;

        // close the connection and return Ok()
        self.database.close_connection(conn)?;
        return Ok(num_updated);
    }

    
    // ========================= Utility Functions ========================= //
    /// Completely erases the message database. Returns an empty tuple on
    /// success and a `std::io::Error` on failure
    pub fn wipe(&self) -> Result<(), std::io::Error>
    {
        std::fs::remove_file(self.file_path.clone())?;
        return Ok(());
    }

}
