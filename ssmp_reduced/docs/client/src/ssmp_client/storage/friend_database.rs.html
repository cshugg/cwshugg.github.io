// A module to hold the struct definitions for interacting with the friend
// database. Utilizes `database.rs`.

// Library/module inclusions
use crate::storage::database;
use database::LocalDatabase;
use database::SQLColumnInfo;
use rusqlite::{ToSql, Connection, Error, NO_PARAMS};

// =============================== Constants =============================== //
#[allow(dead_code)]
const FRIEND_DATABASE_FILE: &str = "friends.db";


// ============================= Friend Schema ============================= //
/// # The Friend Schema
/// A schema used to organize the data kept in the SQLite friend database. The
/// struct fields are as follows:
/// 
/// - `username` - the friend's username
/// - `pubkey` - the friend's public key used to encrypt messages
/// - `ip_addr` - the friend's last known IP address
/// - `message_database_table_name` - the name of the table within the message
///   database used to store this friend's messages
#[derive(Clone)]
pub struct FriendSchema
{
    pub username: (String, SQLColumnInfo),
    pub pubkey: (String, SQLColumnInfo),
    pub ip_addr: (String, SQLColumnInfo),
    pub message_database_table_name: (String, SQLColumnInfo),
}

#[allow(dead_code)]
impl FriendSchema
{
    /// **Constructor**: takes in a username and creates a new default
    /// `FriendSchema` instance.
    pub fn new(user_name: &str) -> FriendSchema
    {
        // create the correct name for the message database table
        let mdbt_name: String = FriendSchema::make_message_database_table_name(user_name);
        return FriendSchema
        {
            username: (user_name.to_string(), SQLColumnInfo::new("username", "MEDIUMTEXT", true)),
            pubkey: ("".to_string(), SQLColumnInfo::new("pubkey", "MEDIUMTEXT", false)),
            ip_addr: ("".to_string(), SQLColumnInfo::new("ip_address", "TINYTEXT", false)),
            message_database_table_name: (mdbt_name, SQLColumnInfo::new("message_database_table", "TINYTEXT", true))
        };
    }


    // ------------------------- Utility Functions ------------------------- //
    /// Uses the schema's current `username` field to update its
    /// `message_database_table_name` field appropriately. This will be called
    /// automatically when interacting with the database.
    pub fn update_message_database_table_name(&mut self)
    {
        let name = FriendSchema::make_message_database_table_name(&self.username.0);
        self.message_database_table_name.0 = name;
    }

    /// Takes in a username and creates the name of the table to create and use
    /// within the message database, in which messages from this friend will be
    /// stored.
    fn make_message_database_table_name(username: &str) -> String
    {
        return format!("friend_{}", username);
    }

    /// Creates a vector of `ToSql` values for SQL commands to use as params
    /// when dealing with this schema. The vector contains every struct field
    /// (except the ones related to SQL itself, such as
    /// `message_database_table_name`) in the order they appear.
    #[allow(dead_code)]
    pub fn to_sql_params(&self) -> Vec<&dyn ToSql>
    {
        let result: Vec<&dyn ToSql> = vec![&self.username.0 as &dyn ToSql,
                                           &self.pubkey.0 as &dyn ToSql,
                                           &self.ip_addr.0 as &dyn ToSql,
                                           &self.message_database_table_name.0 as &dyn ToSql];
        return result;
    }

    /// Creates and returns a string representation of a `FriendSchema`
    #[allow(dead_code)]
    pub fn to_string(&self) -> String
    {
        let mut result: String = String::new();
        // push all struct fields onto the string
        result.push_str(&format!("{:<19}{}\n", "Username:", self.username.0));
        result.push_str(&format!("{:<19}{}\n", "Public Key:", self.pubkey.0));
        result.push_str(&format!("{:<19}{}\n", "IP Address:", self.ip_addr.0));
        result.push_str(&format!("{:<19}{}\n", "Message DB Table:",
                                 self.message_database_table_name.0));
        return result;
    }

    /// Creates and returns a vector containing the SQLColumnInfo structs for
    /// each struct field (in the order they appear).
    pub fn get_sql_column_info(&self) -> Vec<SQLColumnInfo>
    {
        let mut result: Vec<SQLColumnInfo> = Vec::new();
        result.push(self.username.1.clone());
        result.push(self.pubkey.1.clone());
        result.push(self.ip_addr.1.clone());
        result.push(self.message_database_table_name.1.clone());
        return result;
    }
}


// ============================ Friend Database ============================ //
/// # The Friend Database
/// This `FriendDatabase` struct defines an interface used to interact with a
/// SQLite database that stores `FriendSchema`.
/// 
/// Each `FriendDatabase` struct created using its `new()` function will all
/// connect to the same database - the one located within `.ssmp/data/`.
#[allow(dead_code)]
pub struct FriendDatabase
{
    file_path: String,              // path to the .db file
    database: LocalDatabase,        // reference to a LocalDatabase object
    schema: FriendSchema            // a schema instance, for convenience
}

#[allow(dead_code)]
impl FriendDatabase
{
    /// **Constructor**: creates a new `FriendDatabase` struct pointing at the
    /// friend database in the default location (.ssmp/data/friends.db)
    pub fn new() -> FriendDatabase
    {
        // first, create/get the path to the database
        let path_create = database::make_database_path(FRIEND_DATABASE_FILE);
        if path_create.is_err() // for now, panic on error
        { panic!("Could not get/create SSMP database directory!"); }
        let path: String = path_create.unwrap();

        // return a FriendDatabase object
        return FriendDatabase
        {
            file_path: path.clone(),
            database: LocalDatabase::new(&path),
            schema: FriendSchema::new("")
        }
    }

    /// **Constructor**: creates a new `FriendDatabase` struct pointed at the
    /// location specified by the input parameter (`location`).
    pub fn new_at_location(location: &str) -> FriendDatabase
    {
        return FriendDatabase
        {
            file_path: location.to_string(),
            database: LocalDatabase::new(location),
            schema: FriendSchema::new("")
        };
    }

    // ======================= Database Interaction ======================== //
    // --------------- Table Interaction --------------- //
    /// Attempts to create a new table within the friend database. The table's
    /// name is specified by the input parameter, `table_name`. Returns the
    /// command string executed on success, and a `rusqlite::Error` on failure.
    pub fn create_table(&self, table_name: &str) -> Result<String, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // get the schema's column info and pass it to the database.rs
        // function to build a CREATE TABLE command string
        let column_info: Vec<SQLColumnInfo> = self.schema.get_sql_column_info();
        let cmd: String = database::make_create_table_string(table_name, column_info);

        // execute the command
        conn.execute(&cmd, NO_PARAMS)?;
        // close database connection and return the command string
        self.database.close_connection(conn)?;
        return Ok(cmd);
    }

    /// Attempts to drop a table, specified by `table_name`, from the Friend
    /// Database. On success, the command string that was executed is returned,
    /// and on failure, a `rusqlite::Error` is returned.
    pub fn drop_table(&self, table_name: &str) -> Result<String, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // build the DROP TABLE command and execute it
        let cmd: String = database::make_drop_table_string(table_name);
        conn.execute(&cmd, NO_PARAMS)?;

        // close the database connection and return the command string
        self.database.close_connection(conn)?;
        return Ok(cmd);
    }

    // --------------- Inserting Friends --------------- //
    /// Attempts to insert a given friend (`friend`) into the friend database's
    /// table named `table_name`. Returns the executed command string on
    /// success and a `rusqlite::Error` on failure.
    pub fn insert_friend(&self, table_name: &str, friend: FriendSchema) -> Result<String, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // create the command string with the `FriendSchema`'s column info
        let cinfo: Vec<SQLColumnInfo> = friend.get_sql_column_info();
        let cmd: String = database::make_insert_string(table_name, cinfo);
        
        // execute the command
        conn.execute(&cmd, &friend.to_sql_params())?;
        // close the connection and return the command string
        self.database.close_connection(conn)?;
        return Ok(cmd);
    }

    // --------------- Inserting Friends --------------- //
    /// Searches the database for any friends whose username matches the given
    /// username (`name`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count = -1`.)
    pub fn find_friend_by_username(&self, table_name: &str, max_count: i32, name: &str) -> Result<Vec<FriendSchema>, Error>
    { return self.find_friend(table_name, max_count, &self.schema.username.1.name, &name as &dyn ToSql); }
    
    /// Searches the database for any friends whose public key matches the
    /// given key (`pubkey`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count = -1`.)
    pub fn find_friend_by_pubkey(&self, table_name: &str, max_count: i32, pubkey: &str) -> Result<Vec<FriendSchema>, Error>
    { return self.find_friend(table_name, max_count, &self.schema.pubkey.1.name, &pubkey as &dyn ToSql); }
    
    /// Searches the database for any friends whose IP address matches the
    /// given address (`addr`). If more than `max_count` results are found,
    /// only `max_count` will be returned.
    /// (If you want no limit, set `max_count = -1`.)
    pub fn find_friend_by_ip_address(&self, table_name: &str, max_count: i32, addr: &str) -> Result<Vec<FriendSchema>, Error>
    { return self.find_friend(table_name, max_count, &self.schema.ip_addr.1.name, &addr as &dyn ToSql); }


    /// Private helper function that searches the friend database using ANY
    /// FriendSchema field. The field used to search must be properly casted to
    /// a `&dyn ToSql` before being passed in. For example:
    /// 
    /// To search by username:
    /// 
    /// ```
    /// find_friend(table_name, max_count, "username",
    ///             &schema.username.0 as &dyn ToSql);
    /// ```
    /// 
    /// If more than `max_count` results are found, only the first `max_count`
    /// results will be returned in the vector. If an error occurs, a
    /// `rusqlite::Error` is returned instead.
    fn find_friend(&self, table_name: &str, max_count: i32,
                   field_name: &str, val: &dyn ToSql) -> Result<Vec<FriendSchema>, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // put together a command string
        let condition: String = format!("{} = ?1", field_name);
        let cmd: String = database::make_select_string(table_name, &condition,
                                                       self.schema.get_sql_column_info());

        // prepare and execute a query with the command
        let mut command = conn.prepare(&cmd)?;
        let iterator = command.query_map(&[&val], |row|
        {
            // set up a new friend schema with the current row
            let name: String = row.get(0)?;
            let mut fs: FriendSchema = FriendSchema::new(&name);
            fs.pubkey.0 = row.get(1)?;
            fs.ip_addr.0 = row.get(2)?;
            // return the schema into the iterator for future use
            return Ok(fs);
        })?;

        // next, take the iterator and pull out the FriendSchemas to build our
        // resulting FriendSchema vector
        let mut result: Vec<FriendSchema> = Vec::new();
        let mut schema_count: i32 = 0;
        for fs in iterator
        {
            result.push(fs.unwrap());
            schema_count += 1;

            // if we're at max_count, break out of the loop
            if max_count > 0 && schema_count == max_count
            { break; }
        }

        // return the resulting vector
        return Ok(result);
    }

    /// Queries for EVERY entry in the given table. Returns a result containing
    /// either an array of `FriendSchema` or a `rusqlite::Error`.
    pub fn find_all(&self, table_name: &str, max_count: i32) -> Result<Vec<FriendSchema>, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // prepare and execute a query with the command
        let cmd: String = format!("SELECT * FROM {}", table_name);
        let mut command = conn.prepare(&cmd)?;
        let iterator = command.query_map(NO_PARAMS, |row|
        {
            // set up a new friend schema with the current row
            let name: String = row.get(0)?;
            let mut fs: FriendSchema = FriendSchema::new(&name);
            fs.pubkey.0 = row.get(1)?;
            fs.ip_addr.0 = row.get(2)?;
            // return the schema into the iterator for future use
            return Ok(fs);
        })?;

        // next, take the iterator and pull out the FriendSchemas to build our
        // resulting FriendSchema vector
        let mut result: Vec<FriendSchema> = Vec::new();
        let mut schema_count: i32 = 0;
        for fs in iterator
        {
            result.push(fs.unwrap());
            schema_count += 1;

            // if we're at max_count, break out of the loop
            if max_count > 0 && schema_count == max_count
            { break; }
        }

        // return the resulting vector
        return Ok(result);
    }

    // --------------- Deleting Friends ---------------- //
    /// Searches the database for any friends whose username matches the given
    /// username (`name`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count = -1`.)
    pub fn delete_friend_by_username(&self, table_name: &str, name: &str) -> Result<usize, Error>
    { return self.delete_friend(table_name, &self.schema.username.1.name, &name as &dyn ToSql); }
    
    /// Searches the database for any friends whose public key matches the
    /// given key (`pubkey`). If more than `max_count` results are found, only
    /// `max_count` will be returned.
    /// (If you want no limit, set `max_count = -1`.)
    pub fn delete_friend_by_pubkey(&self, table_name: &str, pubkey: &str) -> Result<usize, Error>
    { return self.delete_friend(table_name, &self.schema.pubkey.1.name, &pubkey as &dyn ToSql); }
    
    /// Searches the database for any friends whose IP address matches the
    /// given address (`addr`). If more than `max_count` results are found,
    /// only `max_count` will be returned.
    /// (If you want no limit, set `max_count = -1`.)
    pub fn delete_friend_by_ip_address(&self, table_name: &str, addr: &str) -> Result<usize, Error>
    { return self.delete_friend(table_name, &self.schema.ip_addr.1.name, &addr as &dyn ToSql); }


    /// Private helper function that deletes from the database using ANY
    /// FriendSchema field. The field used to search must be properly casted to
    /// a `&dyn ToSql` before being passed in. For example:
    /// 
    /// To delete by username:
    /// 
    /// ```
    /// delete_friend(table_name, "username",
    ///               &schema.username.0 as &dyn ToSql);
    /// ```
    /// 
    /// On success, the number of deleted entries is returned. On failure, a
    /// `rusqlite::Error` is returned.
    fn delete_friend(&self, table_name: &str, field_name: &str,
                     val: &dyn ToSql) -> Result<usize, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // craft a command string with the given info
        let condition: String = format!("{} = ?1", field_name);
        let cmd: String = database::make_delete_string(table_name, &condition);
        // execute the command (with `val` as the parameter)
        let num_deleted = conn.execute(&cmd, &[&val])?;

        // close the connection and return the number of elements deleted
        self.database.close_connection(conn)?;
        return Ok(num_deleted);
    }
    
    // --------------- Updating Friends ---------------- //
    /// Takes in a username and a new public key and attempts to update the
    /// friend with the given `username`'s `pubkey` field.
    /// Returns the number of updated entries on success and a
    /// `rusqlite::Error` on failure.
    pub fn update_friend_pubkey(&self, table_name: &str, username: &str, new_key: &str) -> Result<usize, Error>
    {
        return self.update_friend(table_name, vec![&self.schema.pubkey.1.name],
                                  vec![&new_key as &dyn ToSql],
                                  &self.schema.username.1.name,
                                  &username as &dyn ToSql);
    }
    
    /// Takes in a username and a new IP address and attempts to update the
    /// friend with the given `username`'s `ip_address` field.
    /// Returns the number of updated entries on success and a
    /// `rusqlite::Error` on failure.
    pub fn update_friend_ip_address(&self, table_name: &str, username: &str, new_addr: &str) -> Result<usize, Error>
    {
        return self.update_friend(table_name, vec![&self.schema.ip_addr.1.name],
                                  vec![&new_addr as &dyn ToSql],
                                  &self.schema.username.1.name,
                                  &username as &dyn ToSql);
    }

    /// Private helper function that takes in values to update entries within
    /// the database. The parameters are as follows:
    /// 
    /// - `table_name` - the name of the table to update
    /// - `update_cols` - the names of the columns in the selected entries that
    ///   will be updated.
    /// - `update_vals` - a vector of `&dyn ToSql` values used to update the
    ///   column values corresponding to `update_cols`
    /// - `field_name` - the field used to narrow down search results
    /// - `val` - the corresponding `&dyn ToSql` value with `field_name` used
    ///   to narrow down search results.
    /// 
    /// Returns the number of entries updated on success, and a
    /// `rusqlite::Error` on failure.
    fn update_friend(&self, table_name: &str, update_cols: Vec<&str>,
                     update_vals: Vec<&dyn ToSql>, field_name: &str,
                     val: &dyn ToSql) -> Result<usize, Error>
    {
        let conn: Connection = self.database.get_connection()?; // connect

        // create the condition substring for the command string
        let condition: String = format!("{} = ?1", field_name);

        // make sure the two given vectors are the same length - return an
        // error otherwise
        if update_cols.len() != update_vals.len()
        { return Err(Error::InvalidParameterCount(update_cols.len(), update_vals.len())); }

        // otherwise, build the 'update' portion of the string
        let mut update: String = String::new();
        for i in 0..update_cols.len()
        {
            // since our 'condition' string used up "?1", these "?X" values
            // must start at "?2"
            update.push_str(&format!("{} = ?{}" , update_cols[i], i + 2));
            // add a comma if necessary
            if i < update_cols.len() - 1
            { update.push_str(", "); }
        }

        // make a combined vector of `val` and the contents of `update_vals`
        let mut arg_vec: Vec<&dyn ToSql> = vec![&val];
        arg_vec.append(&mut update_vals.clone());

        // build the final command string and execute it
        let cmd: String = format!("UPDATE {} SET {} WHERE {};",
                                  table_name, update, condition);
        let num_updated = conn.execute(&cmd, &arg_vec)?;

        // close connection and return
        self.database.close_connection(conn)?;
        return Ok(num_updated);
    }

    
    // ========================= Utility Functions ========================= //
    /// Completely erases the friend database. Returns an empty tuple on
    /// success and a `std::io::Error` on failure
    pub fn wipe(&self) -> Result<(), std::io::Error>
    {
        std::fs::remove_file(self.file_path.clone())?;
        return Ok(());
    }

    // Tests inserting messages into the friend database
    #[test]
    fn fdb_test2_inserting()
    {
        println!("\n=================== FriendDatabase Test 2 ====================");
        let fdb: FriendDatabase = FriendDatabase::new_at_location(DATABASE_FILE);

        // create schemas to insert
        let mut fs1: FriendSchema = FriendSchema::new("turambar");
        fs1.ip_addr.0 = "11.33.33.77".to_string();
        fs1.pubkey.0 = "pubkey1".to_string();

        // insert one!
        let r1 = fdb.insert_friend("X", fs1.clone());
        assert!(r1.is_ok());
        println!("----- Table insert 1: success -----\n{}", r1.unwrap());
        // attempt to insert the same one
        let r2 = fdb.insert_friend("X", fs1.clone());
        assert!(r2.is_err());
        println!("----- Table insert 2: repeated MDB table name -----\n{}", r2.err().unwrap());
        // update the username and message database table name, then re-insert
        fs1.username.0 = "glaurung".to_string();
        fs1.update_message_database_table_name();
        fs1.pubkey.0 = "pubkey2".to_string();
        let r3 = fdb.insert_friend("X", fs1.clone());
        assert!(r3.is_ok());
        println!("----- Table insert 3: success -----\n{}", r3.unwrap());

        // insert one more
        fs1.username.0 = "luthien".to_string();
        fs1.update_message_database_table_name();
        fs1.pubkey.0 = "pubkey3".to_string();
        let r4 = fdb.insert_friend("X", fs1.clone());
        assert!(r4.is_ok());
        println!("----- Table insert 4: success -----\n{}", r4.unwrap());
    }
