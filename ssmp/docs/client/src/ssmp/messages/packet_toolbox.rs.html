// ========================== Integer byte vectors ========================= //
/// Takes a 32-bit integer and converts it to an array of 8-bit integers, where
/// each u8 represents one of the four bytes within the `u32`. The byte
/// ordering is as follows:
/// ```
/// // vec[0] = byte[3] (MSB)
/// // vec[1] = byte[2]
/// // vec[2] = byte[1]
/// // vec[3] = byte[0] (LSB)
/// ```
pub fn u32_to_byte_vector(integer: u32) -> Vec<u8> {
    let mut result: Vec<u8> = Vec::with_capacity(4);
    for i in 0..4 {
        result.push((integer >> (24 - (i * 8))) as u8);
    }

    return result;
}

/// Takes in a vector of 4 bytes (no more, no less) and reads them from 0..3 as
/// a big-endian 32-but unsigned integer. The integer is returned.
///
/// **NOTE**: if there's less than 4 bytes in the vector, the function panics.
/// If there's more than 4 bytes, only the first 4 are taken to represent the
/// integer.
pub fn byte_vector_to_u32(bytes: Vec<u8>) -> u32 {
    // check the vector length
    if bytes.len() < 4 {
        panic!("vector must have at least 4 elements!");
    }

    // iterate through the 4 bytes and reconstruct the u32 in big-endian order
    let mut result: u32 = 0;
    for i in 0..4 {
        result |= (bytes[i] as u32) << (24 - (i * 8));
    }
    // return the resulting integer
    return result;
}

// ========================== Integers to strings ========================== //
/// Takes an 8-bit unsigned integer and builds a string containing the binary,
/// hex, and decimal representations of the integer, and returning it.
pub fn u8_to_detailed_string(integer: u8) -> String {
    return format!("{:08b} {:02x} {:<3}", integer, integer, integer);
}

/// Takes a 32-bit unsigned integer and builds a string containing the binary,
/// hex, and decimal representations of the integer, and returning it.
pub fn u32_to_detailed_string(integer: u32) -> String {
    return format!("{:032b} {:08x} {:<10}", integer, integer, integer);
}

/// Takes in an array of bytes (`u8`s) and builds a vector of strings, each
/// representing a single line in an output print. Each string contains the
/// binary AND hex representation of `bytes_per_row` different bytes in the
/// vector. The vector of strings is returned
pub fn byte_vector_to_detailed_string_vector(bytes: &Vec<u8>, bytes_per_row: usize) -> Vec<String> {
    // calculate the number of rows needed
    let mut rows = bytes.len() / bytes_per_row;
    if bytes.len() % bytes_per_row > 0 {
        rows += 1;
    }
    // allocate a new vector of the appropriate size
    let mut result: Vec<String> = Vec::with_capacity(rows);

    // if there aren't any rows to print, return the string as is
    if rows == 0 {
        return result;
    }

    // iterate through each row
    for i in 0..rows {
        // create temporary strings to hold the row's information
        let mut bin_string = String::new();
        let mut hex_string = String::new();
        let mut txt_string = String::new();
        // iterate through each byte in the row
        let row_start = i * bytes_per_row;
        let row_end = row_start + bytes_per_row;
        for index in row_start..row_end {
            // if the current index has reached past the end of the array,
            // append spaces to the strings instead (for formatting)
            if index >= bytes.len() {
                // only do the following if the number of rows is > 1
                if rows > 1 {
                    bin_string.push_str(&format!("{:8}", ""));
                    hex_string.push_str(&format!("{:2}", ""));
                    txt_string.push_str(&format!("{:1}", ""));
                }
            }
            // otherwise, append the byte like normal
            else {
                let byte = bytes[index];
                // append the byte's binary and hex representations to strings
                bin_string.push_str(&format!("{:08b}", byte));
                hex_string.push_str(&format!("{:02x}", byte));
                // if the byte (as a character) is non-printable, replace it
                // with a "." in the txt_string
                let mut byte_char: char = byte as char;
                if byte < 32 {
                    byte_char = '.';
                }
                txt_string.push_str(&format!("{}", byte_char));
            }
        }

        // append these combined strings to the vector
        result.push(format!("{} {} {}", bin_string, hex_string, txt_string));
    }
    return result;
}

#[cfg(test)]
mod packet_toolbox_tests {
    use super::byte_vector_to_u32;
    use super::u32_to_byte_vector;

    // Tests converting u32s to byte vectors
    #[test]
    fn u32_to_byte_vector_test() {
        let int1: u32 = 0x01234567;
        let v1: Vec<u8> = u32_to_byte_vector(int1);
        assert_eq!(v1[0], 0x01);
        assert_eq!(v1[1], 0x23);
        assert_eq!(v1[2], 0x45);
        assert_eq!(v1[3], 0x67);
    }

    // Tests converting a vector of u8s to a u32
    #[test]
    fn byte_vector_to_u32_test() {
        // test 1: a vector with exactly 4 bytes
        let v1: Vec<u8> = vec![1, 2, 3, 4];
        // 00000001 00000010 00000011 00000100 = 16,909,060 as u32
        assert_eq!(byte_vector_to_u32(v1), 16909060);

        // test 2: a vector with more than 4 bytes
        let v2: Vec<u8> = vec![1, 2, 3, 5, 100, 200];
        // 00000001 00000010 00000011 00000101 = 16,909,061 as u32
        assert_eq!(byte_vector_to_u32(v2), 16909061);
    }

    // Tests converting a vector of u8s to a u32 with a vector containing
    // less than 4 bytes. This should panic.
    #[test]
    #[should_panic]
    fn byte_vector_to_u32_test2() {
        // test with a vector of 3 bytes (the function requires 4)
        let v1: Vec<u8> = vec![1, 2, 3];
        byte_vector_to_u32(v1);
    }
}
