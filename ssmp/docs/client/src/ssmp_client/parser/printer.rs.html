// A module that defines an interface for standardized terminal printing.
//
//      0xdadb0d
//      CS 4284: Systems and Networking Capstone
//      Fall 2020

// Library/crate inclusions
extern crate termcolor; // https://docs.rs/termcolor/1.1.1/termcolor/
use std::io::{Write};
use termcolor::{Color, ColorChoice, ColorSpec, StandardStream, WriteColor};


// =============================== Constants =============================== //
const PROMPT_COLOR: Option<Color> = Some(Color::Rgb(141, 112, 255));
/// A constant used in `print_message()` to print out usernames as different
/// colors.
const USERNAME_COLORS: [Option<Color>; 10] = [
    Some(Color::Rgb(255, 110, 110)), Some(Color::Rgb(255, 160, 87)),
    Some(Color::Rgb(255, 216, 61)), Some(Color::Rgb(109, 222, 22)),
    Some(Color::Rgb(26, 196, 26)), Some(Color::Rgb(59, 217, 175)),
    Some(Color::Rgb(49, 216, 235)), Some(Color::Rgb(79, 138, 255)),
    Some(Color::Rgb(181, 97, 255)), Some(Color::Rgb(207, 37, 201))
];


// ============================ Print Functions ============================ //
/// Prints a prompt for the user.
pub fn print_prompt()
{
    print!("(");
    color_print("ssmp", PROMPT_COLOR);
    print!(") ");
}


/// Prints a standardized usage messages. Takes in the message to print and
/// prints it with a special prefix. It also takes in an `action`, which is
/// a short description of what the command will do. `action` can be left as
/// "" if none is to be printed
pub fn print_usage(usage: &str, action: &str)
{
    // blue print
    color_print("Usage", Some(Color::Cyan));
    color_print(": ", None);
    // print the action (if it was specified)
    if !action.eq("")
    { color_print(&format!("[{}] ", action), Some(Color::Magenta)); }
    // print out the usage message
    color_print(&format!("\"{}\"", usage), None);
    print!("\n");
}

/// Prints a standardized error message. Takes in the message to print and
/// prints it with a special prefix.
pub fn print_error(message: &str)
{
    color_print("Error", Some(Color::Red));
    println!(": {}", message);
}

/// Prints a single line of the 'help' command's menu. Takes in the name of a
/// command and its description.
pub fn print_help_menu_item(command_name: &str, command_desc: &str)
{
    color_print(&format!("{:<16}", command_name), Some(Color::Magenta));
    println!("{}", command_desc);
}

/// Takes in a username and prints it a certain color (decided using modular
/// arithmetic).
pub fn print_username(username: &str)
{
    let modulus: u8 = USERNAME_COLORS.len() as u8;
    // take the sender name and sum all of its character's integer values,
    // modulus the length of COLOR_ARRAY
    let mut color_index: u8 = 0;
    for letter in username.chars()
    { color_index = (color_index + (letter as u8)).rem_euclid(modulus); }

    // use the color index to print the username
    let color: Option<Color> = USERNAME_COLORS[color_index as usize].clone();
    color_print(&format!("{}", username), color);
}

/// Prints a single line (representing a send/received message) formatted like
/// so: `<sender_name>: <message>`
pub fn print_message(sender_name: &str, message: &str)
{
    print_username(sender_name);
    println!(": {}", message);
}

/// Inner function that prints out colored text. If the write fails, the text
/// is instead printed using `print!()`
fn color_print(text: &str, color: Option<Color>)
{
    // set up an output stream
    let mut stream = StandardStream::stdout(ColorChoice::Always);

    // attempt to set the color
    let result1 = stream.set_color(ColorSpec::new().set_fg(color));
    // if the color-set failed, print using println
    if result1.is_err()
    { println!("{}", text); }

    // attempt to print
    let result2 = write!(&mut stream, "{}", text);
    // if the write failed, print using println
    if result2.is_err()
    { println!("{}", text); }

    // attempt to reset the stream
    let _result3 = stream.reset();
}
