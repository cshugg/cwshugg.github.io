// Crate inclusions
use crate::storage::friend_database;
use friend_database::FriendSchema;
use friend_database::FriendDatabase;
use ssmp::security;
use ssmp::user::User;
use crate::storage::message_history;
use message_history::MessageHistory;
use sodiumoxide::crypto::box_::*;

// NOTE: for `PublicKey` information, see:
// https://docs.rs/sodiumoxide/0.2.5/sodiumoxide/crypto/box_/curve25519xsalsa20poly1305/struct.PublicKey.html


// ============================= Friend Struct ============================= //
/// # Friends
/// The `Friend` struct defines a single user that's been marked as a friend by
/// the client. It holds username, IP address, public key, and other info.
#[allow(dead_code)]
pub struct Friend
{
    username: String,
    pubkey: Option<PublicKey>,  // sodiumoxide PublicKey struct
    ip_addr: String,

    pub history: MessageHistory, // MessageHistory object
    schema: FriendSchema        // internal `FriendSchema` to insert into DB
}

#[allow(dead_code)]
impl Friend
{
    /// **Constructor**: takes in a username and sets up a new `Friend` struct.
    pub fn new(name: &str) -> Friend
    {
        // create new FriendSchema and MessageHistory objects
        let fs: FriendSchema = FriendSchema::new(name);
        let ms: MessageHistory = MessageHistory::new(
            &fs.message_database_table_name.0, name);
        
        // create a friend struct
        let mut friend = Friend
        {
            username: name.to_string(),
            pubkey: None,
            ip_addr: "0.0.0.0".to_string(),
            history: ms,
            schema: fs
        };
        // set the IP address within the schema
        friend.schema.ip_addr.0 = friend.ip_addr.clone();
        return friend;
    }

    /// Takes in a `FriendSchema` and creates a corresponding `Friend` struct.
    pub fn new_from_schema(schema: &FriendSchema) -> Friend
    {
        let mut friend: Friend = Friend::new(&schema.username.0);
        friend.set_ip_address(&schema.ip_addr.0);
        friend.set_public_key_string(&schema.pubkey.0);
        return friend;
    }

    /// Takes in a SSMP `User` struct and creates a corresponding `Friend`
    /// structure.
    pub fn new_from_user(user: User) -> Friend
    {
        let mut friend: Friend = Friend::new(&user.user_name);
        friend.set_ip_address(&user.ip_port);
        friend.set_public_key(&user.pub_key);
        return friend;
    }


    // -------------------------- Getters/Setters -------------------------- //
    /// Returns a copy of the friend's username.
    pub fn get_username(&self) -> String
    { return self.username.clone(); }

    /// Returns a copy of the friend's public key. Returns `Some(PublicKey)`
    /// when the public key has a value, and `None` otherwise
    pub fn get_public_key(&self) -> Option<PublicKey>
    { return if self.pubkey.is_some() { self.pubkey.clone() } else { None }; }

    /// Returns a copy of the friend's public key, but in encoded string form.
    /// Returns `Some(string)` on success and `None` otherwise
    pub fn get_public_key_string(&self) -> Option<String>
    {
        // if we have a public key, encode it as a string and return it
        return if self.pubkey.is_some()
        { return Some(security::utils::encode_pub_key(self.pubkey.clone().as_ref().unwrap())); }
        else
        { None };
    }

    /// Returns a copy of the friend's IP address.
    pub fn get_ip_address(&self) -> String
    { return self.ip_addr.clone(); }

    /// Returns a copy of the internal `FriendSchema`. Used to interact with
    /// the `FriendDatabase`.
    pub fn get_schema(&self) -> FriendSchema
    { return self.schema.clone(); }

    /// Takes in a new `PublicKey` and updates the `Friend` struct accordingly.
    pub fn set_public_key(&mut self, new_pubkey: &PublicKey)
    {
        self.pubkey = Some(new_pubkey.clone());
        // update the schema's encoded string
        
        self.schema.pubkey.0 = security::utils::encode_pub_key(new_pubkey);
    }

    /// Takes in an encoded public key (as a `&str`) and attempts to decode it
    /// into a `PublicKey` struct. Returns `true` on success and `false` on
    /// failure to decode.
    pub fn set_public_key_string(&mut self, new_pubkey: &str) -> bool
    {
        // if the given string is an empty string (""), do NOT decode it
        if new_pubkey.eq("")
        {
            self.pubkey = None;
            return false;
        }

        // attempt to decode and return `false` on error
        let result = security::utils::decode_pub_key(new_pubkey);
        if result.is_err()
        {
            self.pubkey = None;
            return false;
        }

        // otherwise, pull out the `PublicKey` and set struct fields
        self.pubkey = Some(result.unwrap());
        self.schema.pubkey.0 = new_pubkey.to_string();
        return true;
    }

    /// Takes in a new IP address and updates the `Friend` struct accordingly.
    pub fn set_ip_address(&mut self, new_ip_addr: &str)
    {
        // update both the friend's field and the schema's field
        self.ip_addr = new_ip_addr.to_string();
        self.schema.ip_addr.0 = self.ip_addr.clone();
    }
}


// ========================== Friend List Struct =========================== //
/// # Friend List
/// The `FriendList` struct serves as an interface to the `FriendDatabase`
/// defined in `friend_database.rs`. This struct supports common operations
/// that can be called without directly interacting with the database.
/// 
/// Every `FriendList` struct uses the same database. (Multiple `FriendList`
/// struct instances won't write friends to different database.)
#[allow(dead_code)]
pub struct FriendList
{
    size: u32,
    db_table_names: Vec<String>,
    db: FriendDatabase
}

#[allow(dead_code)]
impl FriendList
{
    /// **Constructor**: creates and returns a `FriendList` pointed at the
    /// default database location.
    pub fn new() -> FriendList
    {
        let list = FriendList
        {
            size: 0,
            db_table_names: vec!["main".to_string()],
            db: FriendDatabase::new()
        };
        // set up the database and return the object
        list.setup_database();
        return list;
    }

    /// **Constructor**: creates and returns a `FriendList` pointed at a
    /// database at the location specified.
    pub fn new_at_location(location: &str) -> FriendList
    {
        let list = FriendList
        {
            size: 0,
            db_table_names: vec!["main".to_string()],
            db: FriendDatabase::new_at_location(location)
        };
        // set up the database and return the object
        list.setup_database();
        return list;
    }
    
    /// Sets up the initial table(s) for `FriendSchema`. Returns `false` if an
    /// error occurred while setting things up.
    fn setup_database(&self) -> bool
    {
        // iterate through the vector of table names and create the tables
        for table_name in &self.db_table_names
        {
            let result = self.db.create_table(table_name);
            if result.is_err()
            { false; }
        }
        return true;
    }


    // -------------------------- CRUD Operations -------------------------- //
    /// Attempts to add the given friend to the friend list. Returns `true` on
    /// success and `false` on internal error.
    pub fn add_friend(&self, friend: &Friend) -> bool
    {
        // attempt to insert the friend's schema into the database
        let result = self.db.insert_friend(&self.db_table_names[0], friend.get_schema());
        return result.is_ok();
    }

    /// Searches for ALL friends in the friend list. Returns a vector of Friend
    /// structs on success, and `None` on failure.
    pub fn find_all(&self) -> Option<Vec<Friend>>
    {
        let result = self.db.find_all(&self.db_table_names[0], -1);
        // if an error was returned, return `None`
        if result.is_err()
        { return None; }

        // otherwise, pull out the `FriendSchema` vector.
        // If the vector is empty, then nothing was found.
        let schema: Vec<FriendSchema> = result.unwrap();
        if schema.len() == 0
        { return None; }

        // otherwise, iterate through the `FriendSchema` and build a vector of
        // `Friend` structs
        let mut friends: Vec<Friend> = Vec::with_capacity(schema.len());
        for fs in schema
        { friends.push(Friend::new_from_schema(&fs)); }
        return Some(friends);
    }

    /// Searches the friend list for a friend with a matching username. If one
    /// is found, a `Friend` struct is returned. Otherwise, `None` is returned.
    pub fn find_by_username(&self, username: &str) -> Option<Friend>
    {
        let result = self.db.find_friend_by_username(&self.db_table_names[0], 1, username);
        // if an error was returned, return `None`
        if result.is_err()
        { return None; }

        // otherwise, pull out the `FriendSchema` and build a `Friend` struct.
        // If the vector is empty, then nothing was found. Otherwise, return
        // the first vector entry
        let schema: Vec<FriendSchema> = result.unwrap();
        if schema.len() == 0
        { return None; }
        return Some(Friend::new_from_schema(&schema[0]));
    }

    /// Searches the friend list for a friend with a matching public key. If one
    /// is found, a `Friend` struct is returned. Otherwise, `None` is returned.
    pub fn find_by_public_key(&self, pubkey: &PublicKey) -> Option<Friend>
    {
        // convert the public key to a string
        let key_string: String = security::utils::encode_pub_key(pubkey);
        let result = self.db.find_friend_by_pubkey(&self.db_table_names[0], 1, &key_string);
        // if an error was returned, return `None`
        if result.is_err()
        { return None; }

        // otherwise, pull out the `FriendSchema` and build a `Friend` struct.
        // If the vector is empty, then nothing was found. Otherwise, return
        // the first vector entry
        let schema: Vec<FriendSchema> = result.unwrap();
        if schema.len() == 0
        { return None; }
        return Some(Friend::new_from_schema(&schema[0]));
    }

    /// Searches the friend list for friends with a matching IP address. Since
    /// multiple friends could have the same IP address, this function returns
    /// a `Vec<Friend>` on success, rather than just a `Friend`. On failure, it
    /// returns `None`.
    pub fn find_by_ip_address(&self, address: &str) -> Option<Vec<Friend>>
    {
        let result = self.db.find_friend_by_ip_address(&self.db_table_names[0], -1, address);
        // if an error was returned, return `None`
        if result.is_err()
        { return None; }

        // otherwise, pull out the `FriendSchema` and build a `Friend` struct.
        // If the vector is empty, then nothing was found.
        let schema: Vec<FriendSchema> = result.unwrap();
        if schema.len() == 0
        { return None; }

        // otherwise, iterate through the `FriendSchema` and build a vector of
        // `Friend` structs
        let mut friends: Vec<Friend> = Vec::with_capacity(schema.len());
        for fs in schema
        { friends.push(Friend::new_from_schema(&fs)); }
        return Some(friends);
    }

    /// Takes in a username and `PublicKey` and attempts to update the entry in
    /// the friend list with the matching username. Returns `true` on success
    /// and `false` on failure.
    pub fn update_friend_public_key(&self, username: &str, new_key: &PublicKey) -> bool
    {
        // create a temporary `Friend` to store and convert the PublicKey into
        // a string, to be inserted into the database
        let mut temp: Friend = Friend::new("temp");
        temp.set_public_key(&new_key);
        let key_string = temp.get_public_key_string();
        // if the converted string is `None`, then an error occurred while
        // converting. Return `false`
        if key_string.is_none()
        { return false; }

        // invoke the `FriendDatabase` 
        let result = self.db.update_friend_pubkey(&self.db_table_names[0],
                                                  username, &key_string.unwrap());
        // if the result is an error, return false
        if result.is_err()
        { return false; }
        // otherwise, check the return value (the number of entries updated).
        // return `false` if it's 0
        return result.unwrap() > 0;
    }

    /// Takes in a username and IP address (both as strings) and attempts to
    /// update the entry in the friend list with the matching username. Returns
    /// `true` on success and `false` on failure.
    pub fn update_friend_ip_address(&self, username: &str, new_addr: &str) -> bool
    {
        // invoke the `FriendDatabase` and return the result's `.is_ok()`
        let result = self.db.update_friend_ip_address(&self.db_table_names[0],
                                                      username, new_addr);
        // if the result is an error, return false
        if result.is_err()
        { return false; }
        // otherwise, check the value that was returned. If it's zero (meaning
        // zero entries were updated), return false
        return result.unwrap() > 0;
    }

    /// Attempts to remove the friend with the given username from the friend
    /// list. Returns `true` on a successful removal and `false` otherwise.
    pub fn remove_friend(&self, username: &str) -> bool
    {
        // search the friend list for the friend
        let fresult = self.find_by_username(username);
        if fresult.is_none()
        { return false; }
        let friend: Friend = fresult.unwrap();
        
        // attempt to delete the friend from the friend database
        let result = self.db.delete_friend_by_username(&self.db_table_names[0],
                                                       &friend.username);
        // if the result is an error, return false
        if result.is_err()
        { return false; }
        // if nothing was actually deleted, return false
        if result.unwrap() <= 0
        { return false; }

        // finally, use the extracted table name to delete the corresponding
        // table from the message database. Returns whether or not this
        // succeeds
        return friend.history.wipe();

    }

    /// Attempts to clear the entire friend list. (The SQLite database used to
    /// store the friends is completely deleted.) Returns `true` on a successful
    /// wipe and `false` on a failure.
    pub fn wipe(&self) -> bool
    {
        // first, extra all friends from the database
        let all_result = self.find_all();
        if all_result.is_none()
        { return false; }
        // extract vector of friends
        let friends: Vec<Friend> = all_result.unwrap();

        // iterate through each friend and wipe his/her message history
        for friend in friends
        {
            if !friend.history.wipe()
            { println!("Internal friend list error: could not wipe '{}'s messages.", friend.username); }
        }

        // finally, wipe the friend database itself, and return whether ot not
        // the wipe was a success
        let wipe_ok = self.db.wipe().is_ok();
        if !wipe_ok
        { return false; }
        // if the wipe was successful, re-create the tables
        self.setup_database();
        return wipe_ok;
    }
}


// ============================== Unit Tests =============================== //
#[cfg(test)]
mod friend_tests
{
    use super::Friend;
    use ssmp::security;
    use sodiumoxide::crypto::box_::*;

    // Tests the Friend struct's getters and setters
    #[test]
    fn test1_getters_setters()
    {
        println!("\n==================== Friend Test 1 ====================");
        let mut f1: Friend = Friend::new("user1");
        let (pubk, _privk) = security::utils::gen_pub_priv_keypair();
        f1.set_public_key(&pubk);
        println!("User 1:\n{}", f1.get_schema().to_string());

        // assert default values
        assert_eq!(f1.get_username(), "user1");
        assert_eq!(f1.get_ip_address(), "0.0.0.0");
        assert_eq!(f1.get_schema().username.0, "user1");
        assert_eq!(f1.get_schema().ip_addr.0, "0.0.0.0");
        assert_eq!(f1.get_schema().message_database_table_name.0, "friend_user1");
        assert_eq!(f1.get_public_key_string().unwrap(), security::utils::encode_pub_key(&pubk));
        
        // set ip address and re-check
        f1.set_ip_address("1.2.3.4");
        assert_eq!(f1.get_ip_address(), "1.2.3.4");
        assert_eq!(f1.get_schema().ip_addr.0, "1.2.3.4");

        // set public key and re-check
        let pk: PublicKey = f1.get_public_key().unwrap();
        let (pubk, _privk) = security::utils::gen_pub_priv_keypair();
        f1.set_public_key(&pubk);
        assert!(!f1.get_schema().pubkey.0.eq(&security::utils::encode_pub_key(&pk)));

        // set public key as a string and re-check
        let (pubk, _privk) = security::utils::gen_pub_priv_keypair();
        let pubk_string = security::utils::encode_pub_key(&pubk);
        f1.set_public_key_string(&pubk_string);
        assert!(f1.get_schema().pubkey.0.eq(&pubk_string));

    }
}

#[cfg(test)]
mod friendlist_tests
{
    use super::Friend;
    use super::FriendList;
    use crate::storage::friend_database;
    use ssmp::security;
    
    const DATABASE_FILE: &str = "fdb2.db";

    // Tests the Friend struct's getters and setters
    #[test]
    fn test2_add_search()
    {
        println!("\n================== FriendList Test 2 ==================");
        let fl: FriendList = FriendList::new_at_location(DATABASE_FILE);

        // create a friend and add it to the database
        let mut friend1 = Friend::new("friend1");
        let (pubk, _privk) = security::utils::gen_pub_priv_keypair();
        friend1.set_public_key(&pubk);
        assert!(fl.add_friend(&friend1));

        // test a failed search
        let result0 = fl.find_by_username("this_user_does_not_exist");
        assert!(result0.is_none());

        // search for the friend by its username
        let result1 = fl.find_by_username(&friend1.get_username());
        assert!(result1.is_some());
        let result1_unwrap: Friend = result1.unwrap();
        assert_eq!(friend1.get_username(), result1_unwrap.get_username());
        assert!(friend1.get_public_key().eq(&result1_unwrap.get_public_key()));
        assert_eq!(friend1.get_ip_address(), result1_unwrap.get_ip_address());

        // search for the friend by its pubkey
        let result2 = fl.find_by_public_key(friend1.get_public_key().as_ref().unwrap());
        assert!(result2.is_some());
        let result2_unwrap: Friend = result2.unwrap();
        assert_eq!(friend1.get_username(), result2_unwrap.get_username());
        assert!(friend1.get_public_key().eq(&result2_unwrap.get_public_key()));
        assert_eq!(friend1.get_ip_address(), result2_unwrap.get_ip_address());

        // search for the friend by its ip address
        let result3 = fl.find_by_ip_address(&friend1.get_ip_address());
        assert!(result3.is_some());
        let result3_unwrap: Vec<Friend> = result3.unwrap();
        assert!(result3_unwrap.len() == 1);
        assert_eq!(friend1.get_username(), result3_unwrap[0].get_username());
        assert!(friend1.get_public_key().eq(&result3_unwrap[0].get_public_key()));
        assert_eq!(friend1.get_ip_address(), result3_unwrap[0].get_ip_address());

        // search for all friends
        let result4 = fl.find_all();
        assert!(result4.is_some());
        let result4_unwrap: Vec<Friend> = result4.unwrap();
        assert!(result4_unwrap.len() == 1);
    }

    #[test]
    fn test3_update()
    {
        println!("\n================== FriendList Test 3 ==================");
        let fl: FriendList = FriendList::new_at_location(DATABASE_FILE);

        let friend1_search1 = fl.find_by_username("friend1");
        assert!(friend1_search1.is_some());
        let friend1 = friend1_search1.unwrap();

        // attempt to update a non-existent friend
        let result1 = fl.update_friend_ip_address("doesnt_exist", "7.7.7.7");
        assert!(!result1);
        let (new_pubk, _privk) = security::utils::gen_pub_priv_keypair();
        let result2 = fl.update_friend_public_key("doesnt_exist", &new_pubk);
        assert!(!result2);
        // make sure nothing was updated
        let friend1_search2 = fl.find_by_username("friend1");
        assert!(friend1_search2.is_some());
        let friend2 = friend1_search2.unwrap();
        assert_eq!(friend1.get_ip_address(), friend2.get_ip_address());
        assert!(friend1.get_public_key().eq(&friend2.get_public_key()));

        // update 'friend1's IP address and verify the update
        let result3 = fl.update_friend_ip_address("friend1", "8.8.8.8");
        assert!(result3);
        let friend1_search3 = fl.find_by_username("friend1");
        assert!(friend1_search3.is_some());
        let friend3 = friend1_search3.unwrap();
        assert_eq!(friend3.get_ip_address(), "8.8.8.8");

        // update 'friend1's public key and verify the update
        let result4 = fl.update_friend_public_key("friend1", &new_pubk);
        assert!(result4);
        let friend1_search4 = fl.find_by_username("friend1");
        assert!(friend1_search4.is_some());
        let friend4 = friend1_search4.unwrap();
        assert!(friend4.get_public_key().unwrap().eq(&new_pubk));
    }
